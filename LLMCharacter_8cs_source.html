166 void AssignLLM()
167 {
168 if (remote || llm != null) return;
169
170 LLM[] existingLLMs = FindObjectsOfType<LLM>();
171 if (existingLLMs.Length == 0) return;
172
173 SortBySceneAndHierarchy(existingLLMs);
174 llm = existingLLMs[0];
175 string msg = $"Assigning LLM {llm.name} to LLMCharacter {name}";
176 if (llm.gameObject.scene != gameObject.scene) msg += $" from scene {llm.gameObject.scene}";
177 LLMUnitySetup.Log(msg);
178 }

180 void SortBySceneAndHierarchy(LLM[] array)
181 {
182 for (int i = 0; i < array.Length - 1; i++)
183 {
184 bool swapped = false;
185 for (int j = 0; j < array.Length - i - 1; j++)
186 {
187 bool sameScene = array[j].gameObject.scene == array[j + 1].gameObject.scene;
188 bool swap = (
189 (!sameScene && array[j + 1].gameObject.scene == gameObject.scene) ||
190 (sameScene && array[j].transform.GetSiblingIndex() > array[j + 1].transform.GetSiblingIndex())
191 );
192 if (swap)
193 {
194 LLM temp = array[j];
195 array[j] = array[j + 1];
196 array[j + 1] = temp;
197 swapped = true;
198 }
199 }
200 if (!swapped) break;
201 }

456 public async Task<string> Chat(string query, Callback<string> callback = null, EmptyCallback completionCallback = null, bool addToHistory = true)
457 {
458 // handle a chat message by the user
459 // call the callback function while the answer is received
460 // call the completionCallback function when the answer is fully received
461 await LoadTemplate();
462 await InitNKeep();
463
464 string json;
465 await chatLock.WaitAsync();
466 try
467 {
468 AddPlayerMessage(query);
469 string prompt = template.ComputePrompt(chat, AIName);
470 json = JsonUtility.ToJson(GenerateRequest(prompt));
471 chat.RemoveAt(chat.Count - 1);
472 }
473 finally
474 {
475 chatLock.Release();
476 }
477
478 string result = await CompletionRequest(json, callback);
479
480 if (addToHistory && result != null)
481 {
482 await chatLock.WaitAsync();
483 try
484 {
485 AddPlayerMessage(query);
486 AddAIMessage(result);
487 }
488 finally
489 {
490 chatLock.Release();
491 }
492 if (save != "") _ = Save(save);
493 }
494
495 completionCallback?.Invoke();
496 return result;
497 }

628 {
629 // template function to convert the json received and get the content

670 protected async Task<Ret> PostRequestLocal<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
671 {
672 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
673 // this function has streaming functionality i.e. handles the answer while it is being received
674 string callResult = null;
675 bool callbackCalled = false;
676 while (!llm.failed && !llm.started) await Task.Yield();
677 switch (endpoint)
678 {
679 case "tokenize":
680 callResult = await llm.Tokenize(json);
681 break;
682 case "detokenize":
683 callResult = await llm.Detokenize(json);
684 break;
685 case "slots":
686 callResult = await llm.Slot(json);
687 break;
688 case "completion":
689 Callback<string> callbackString = null;
690 if (stream && callback != null)
691 {
692 if (typeof(Ret) == typeof(string))
693 {
694 callbackString = (strArg) =>
695 {
696 callback(ConvertContent(strArg, getContent));
697 };
698 }
699 else
700 {
701 LLMUnitySetup.LogError($"wrong callback type, should be string");
702 }
703 callbackCalled = true;
704 }
705 callResult = await llm.Completion(json, callbackString);
706 break;
707 default:
708 LLMUnitySetup.LogError($"Unknown endpoint {endpoint}");
709 break;
710 }
711
712 Ret result = ConvertContent(callResult, getContent);
713 if (!callbackCalled) callback?.Invoke(result);
714 return result;
715 }

717 protected async Task<Ret> PostRequestRemote<Res, Ret>(string json, string endpoint, ContentCallback<Res, Ret> getContent, Callback<Ret> callback = null)
718 {
719 // send a post request to the server and call the relevant callbacks to convert the received content and handle it
720 // this function has streaming functionality i.e. handles the answer while it is being received
721 if (endpoint == "slots")
722 {
723 LLMUnitySetup.LogError("Saving and loading is not currently supported in remote setting");
724 return default;
725 }
726
727 Ret result = default;
728 byte[] jsonToSend = new System.Text.UTF8Encoding().GetBytes(json);
729 using (var request = UnityWebRequest.Put($"{host}:{port}/{endpoint}", jsonToSend))
730 {
731 WIPRequests.Add(request);
732
733 request.method = "POST";
734 if (requestHeaders != null)
735 {
736 for (int i = 0; i < requestHeaders.Count; i++)
737 request.SetRequestHeader(requestHeaders[i].Item1, requestHeaders[i].Item2);
738 }
739
740 // Start the request asynchronously
741 var asyncOperation = request.SendWebRequest();
742 float lastProgress = 0f;
743 // Continue updating progress until the request is completed
744 while (!asyncOperation.isDone)
745 {
746 float currentProgress = request.downloadProgress;
747 // Check if progress has changed
748 if (currentProgress != lastProgress && callback != null)
749 {
750 callback?.Invoke(ConvertContent(request.downloadHandler.text, getContent));
751 lastProgress = currentProgress;
752 }
753 // Wait for the next frame
754 await Task.Yield();
755 }
756 WIPRequests.Remove(request);
757 if (request.result != UnityWebRequest.Result.Success) LLMUnitySetup.LogError(request.error);
758 else result = ConvertContent(request.downloadHandler.text, getContent);
759 callback?.Invoke(result);
760 }
761 return result;
762 }
