72 public GGUFReader(string path, string mode = "r")
73 {
74 // data = new MemoryStream(File.ReadAllBytes(path));
75 data = new FileStream(path, FileMode.Open, FileAccess.Read);
76 int offs = 0;
77
78 if (BitConverter.ToUInt32(ReadBytes(offs, 4), 0) != GGUF_MAGIC)
79 throw new ArgumentException("GGUF magic invalid");
80 offs += 4;
81
82 uint temp_version = BitConverter.ToUInt32(ReadBytes(offs, 4));
83 if ((temp_version & 65535) == 0)
84 {
85 byte[] tempBytes = ReadBytes(offs, 4);
86 Array.Reverse(tempBytes);
87 temp_version = BitConverter.ToUInt32(tempBytes, 0);

90

95 ulong[] temp_counts = new ulong[2];

97 offs += PushField(new ReaderField { offset = offs, name = "GGUF.tensor_count", parts = new List<Array> { new ulong[] { temp_counts[0] } }, data = new List<int> { 0 }, types = new List<GGUFValueType> { GGUFValueType.UINT64 } });

120 private int PushField(ReaderField field, bool skip_sum = false)
121 {
122 if (fields.ContainsKey(field.name))
123 throw new ArgumentException($"Duplicate {field.name} already in list at offset {field.offset}");
124 fields[field.name] = field;
125 if (skip_sum)
126 return 0;
127 int sum = 0;
128 for (int i = 0; i < field.parts.Count; i++)
129 {
130 Type partType = gguf_scalar_to_np[field.types[i]];
131 sum += Marshal.SizeOf(partType) * field.parts[i].Length;
132 }
133 return sum;

143 private (int, List<Array>, List<int>, List<GGUFValueType>) GetFieldParts(int orig_offs, int raw_type)
144 {
145 int offs = orig_offs;
146 List<GGUFValueType> types = new List<GGUFValueType>();
147 types.Add((GGUFValueType)raw_type);
148 // Handle strings.
149 if ((GGUFValueType)raw_type == GGUFValueType.STRING)
150 {
151 (ulong[] slen, byte[] sdata) = GetStr(offs);
152 List<Array> sparts = new List<Array> { slen, sdata };
153 int size = slen.Length * sizeof(ulong) + sdata.Length;
154 return (size, sparts, new List<int> { 1 }, types);
155 }
156
157 // Check if it's a simple scalar type.
158 if (gguf_scalar_to_np.TryGetValue((GGUFValueType)raw_type, out Type nptype))
159 {
160 Array val = ReadBytes(offs, Marshal.SizeOf(nptype));
161 int size = nptype == typeof(bool) ? 1 : Marshal.SizeOf(nptype);
162 return (size, new List<Array> { val }, new List<int> { 0 }, types);
163 }
164
165 // Handle arrays.
166 if ((GGUFValueType)raw_type == GGUFValueType.ARRAY)
167 {
168 int raw_itype = BitConverter.ToInt32(ReadBytes(offs, 4));
169 offs += Marshal.SizeOf(typeof(int));
170
171 ulong alen = BitConverter.ToUInt64(ReadBytes(offs, 8));
172 offs += Marshal.SizeOf(typeof(ulong));
173
174 List<Array> aparts = new List<Array> { BitConverter.GetBytes(raw_itype), BitConverter.GetBytes(alen) };
175 List<int> data_idxs = new List<int>();
176
177 for (int idx = 0; idx < (int)alen; idx++)
178 {
179 (int curr_size, List<Array> curr_parts, List<int> curr_idxs, List<GGUFValueType> curr_types) = GetFieldParts(offs, raw_itype);
180 if (idx == 0)
181 types.AddRange(curr_types);
182
183 int idxs_offs = aparts.Count;
184 aparts.AddRange(curr_parts);
185 data_idxs.AddRange(new List<int>(curr_idxs.ConvertAll(i => i + idxs_offs)));
186 offs += curr_size;
187 }
188 return (offs - orig_offs, aparts, data_idxs, types);
189 }
190 // We can't deal with this one.
191 throw new ArgumentException($"Unknown/unhandled field type {(GGUFValueType)raw_type}");
192 }

194 private int BuildFields(int offs, int count)
195 {
196 for (int i = 0; i < count; i++)
197 {
198 int orig_offs = offs;
199 (ulong[] kv_klen, byte[] kv_kdata) = GetStr(offs);
200 offs += Marshal.SizeOf(typeof(ulong)) + kv_kdata.Length;
201
202 int raw_kv_type = BitConverter.ToInt32(ReadBytes(offs, 4));
203 offs += Marshal.SizeOf(typeof(int));
204 List<Array> parts = new List<Array> { kv_klen, kv_kdata, BitConverter.GetBytes(raw_kv_type) };
205 List<int> idxs_offs = new List<int> { parts.Count };
206
207 (int field_size, List<Array> field_parts, List<int> field_idxs, List<GGUFValueType> field_types) = GetFieldParts(offs, raw_kv_type);
208 if (field_size == -1)
209 continue;
210
211 parts.AddRange(field_parts);
212 ReaderField readerField = new ReaderField
213 {
214 offset = orig_offs,
215 name = System.Text.Encoding.UTF8.GetString(kv_kdata),
216 parts = parts,
217 data = new List<int>(field_idxs.ConvertAll(idx => idx + idxs_offs[0])),
218 types = field_types
219 };
220 PushField(readerField, skip_sum: true);
221 offs += field_size;
222 }
223 return offs;
224 }
