285 public static async Task DownloadFile(
286 string fileUrl, string savePath, bool overwrite = false,
287 Action<string> callback = null, Action<float> progressCallback = null, bool debug = true
288 )
289 {
290 if (File.Exists(savePath) && !overwrite)
291 {
292 if(debug) Log($"File already exists at: {savePath}");
293 }
294 else
295 {
296 if(debug) Log($"Downloading {fileUrl} to {savePath}...");
297 string tmpPath = Path.Combine(Application.temporaryCachePath, Path.GetFileName(savePath));
298
299 ResumingWebClient client = new ResumingWebClient();
300 downloadClients[savePath] = client;
301 if (File.Exists(tmpPath) && overwrite) File.Delete(tmpPath);
302 await client.DownloadFileTaskAsyncResume(new Uri(fileUrl), tmpPath, !overwrite, progressCallback);
303 downloadClients.Remove(savePath);
304#if UNITY_EDITOR
305 AssetDatabase.StartAssetEditing();
306#endif
307 Directory.CreateDirectory(Path.GetDirectoryName(savePath));
308 if (File.Exists(savePath)) File.Delete(savePath);
309 File.Move(tmpPath, savePath);
310#if UNITY_EDITOR
311 AssetDatabase.StopAssetEditing();
312#endif
313 if(debug) Log($"Download complete!");
314 }
315
316 progressCallback?.Invoke(1f);
317 callback?.Invoke(savePath);
318 }

338 public static async Task AndroidExtractFileOnce(string assetName, bool overwrite = false, bool log = true, int chunkSize = 1024 * 1024)
339 {
340 string source = "jar:file://" + Application.dataPath + "!/assets/" + assetName;
341 string target = GetAssetPath(assetName);
342 if (!overwrite && File.Exists(target))
343 {
344 if (log) Log($"File {target} already exists");
345 return;
346 }
347
348 Log($"Extracting {source} to {target}");
349
350 // UnityWebRequest to read the file from StreamingAssets
351 UnityWebRequest www = UnityWebRequest.Get(source);
352 // Send the request and await its completion
353 var operation = www.SendWebRequest();
354
355 while (!operation.isDone) await Task.Delay(1);
356 if (www.result != UnityWebRequest.Result.Success)
357 {
358 LogError("Failed to load file from StreamingAssets: " + www.error);
359 }
360 else
361 {
362 byte[] buffer = new byte[chunkSize];
363 using (Stream responseStream = new MemoryStream(www.downloadHandler.data))
364 using (FileStream fileStream = new FileStream(target, FileMode.Create, FileAccess.Write))
365 {
366 int bytesRead;
367 while ((bytesRead = await responseStream.ReadAsync(buffer, 0, buffer.Length)) > 0)
368 {
369 await fileStream.WriteAsync(buffer, 0, bytesRead);
370 }
371 }
372 }
373 }

391 public static string RelativePath(string fullPath, string basePath)
392 {
393 // Get the full paths and replace backslashes with forward slashes (or vice versa)
394 string fullParentPath = GetFullPath(basePath).TrimEnd('/');
395 string fullChildPath = GetFullPath(fullPath);
396
397 string relativePath = fullChildPath;
398 if (fullChildPath.StartsWith(fullParentPath, StringComparison.OrdinalIgnoreCase))
399 {
400 relativePath = fullChildPath.Substring(fullParentPath.Length);
401 while (relativePath.StartsWith("/")) relativePath = relativePath.Substring(1);
402 }
403 return relativePath;
404 }

428 static void ExtractInsideDirectory(string zipPath, string extractPath, string prefix = "", bool overwrite = true)
429 {
430 using (ZipArchive archive = ZipFile.OpenRead(zipPath))
431 {
432 foreach (ZipArchiveEntry entry in archive.Entries)
433 {
434 if (string.IsNullOrEmpty(entry.Name))
435 continue; // Skip directories
436
437 string destinationPath;
438 if (!String.IsNullOrEmpty(prefix))
439 {
440 string normalizedPath = entry.FullName.Replace('\\', '/');
441 if (!normalizedPath.StartsWith(prefix, StringComparison.OrdinalIgnoreCase))
442 continue;
443 destinationPath = Path.Combine(extractPath, normalizedPath.Substring(prefix.Length));
444 }
445 else
446 {
447 destinationPath = Path.Combine(extractPath, entry.FullName);
448 }
449
450 Directory.CreateDirectory(Path.GetDirectoryName(destinationPath));
451 entry.ExtractToFile(destinationPath, overwrite);
452 }
453 }

597 public static int GetMaxFreqKHz(int cpuId)
598 {
599 string[] paths = new string[]
600 {
601 $"/sys/devices/system/cpu/cpufreq/stats/cpu{cpuId}/time_in_state",
602 $"/sys/devices/system/cpu/cpu{cpuId}/cpufreq/stats/time_in_state",
603 $"/sys/devices/system/cpu/cpu{cpuId}/cpufreq/cpuinfo_max_freq"
604 };
605
606 foreach (var path in paths)
607 {
608 if (!File.Exists(path)) continue;
609
610 int maxFreqKHz = 0;
611 using (StreamReader sr = new StreamReader(path))
612 {
613 string line;
614 while ((line = sr.ReadLine()) != null)
615 {
616 string[] parts = line.Split(' ');
617 if (parts.Length > 0 && int.TryParse(parts[0], out int freqKHz))
618 {
619 if (freqKHz > maxFreqKHz)
620 {
621 maxFreqKHz = freqKHz;
622 }
623 }
624 }
625 }
626 if (maxFreqKHz != 0) return maxFreqKHz;
627 }
628 return -1;
629 }
