145 public static string GetLLMManagerAssetEditor(string path)
146 {
147 // empty
148 if (string.IsNullOrEmpty(path)) return path;
149 // LLMManager - return location the file will be stored in StreamingAssets
150 ModelEntry modelEntry = LLMManager.Get(path);
151 if (modelEntry != null) return modelEntry.filename;
152 // StreamingAssets - return relative location within StreamingAssets
153 string assetPath = LLMUnitySetup.GetAssetPath(path); // Note: this will return the full path if a full path is passed
154 string basePath = LLMUnitySetup.GetAssetPath();
155 if (File.Exists(assetPath))
156 {
157 if (LLMUnitySetup.IsSubPath(assetPath, basePath)) return LLMUnitySetup.RelativePath(assetPath, basePath);
158 }
159 // full path
160 if (!File.Exists(assetPath))
161 {
162 LLMUnitySetup.LogError($"Model {path} was not found.");
163 }
164 else
165 {
166 string errorMessage = $"The model {path} was loaded locally. You can include it in the build in one of these ways:";
167 errorMessage += $"\n-Copy the model inside the StreamingAssets folder and use its StreamingAssets path";
168 errorMessage += $"\n-Load the model with the model manager inside the LLM GameObject and use its filename";
169 LLMUnitySetup.LogWarning(errorMessage);
170 }
171 return path;
172 }

174 public static string GetLLMManagerAssetRuntime(string path)
175 {
176 // empty
177 if (string.IsNullOrEmpty(path)) return path;
178 // LLMManager
179 string managerPath = LLMManager.GetAssetPath(path);
180 if (!string.IsNullOrEmpty(managerPath) && File.Exists(managerPath)) return managerPath;
181 // StreamingAssets
182 string assetPath = LLMUnitySetup.GetAssetPath(path);
183 if (File.Exists(assetPath)) return assetPath;
184 // give up
185 return path;

314 protected virtual string GetLlamaccpArguments()
315 {
316 // Start the LLM server in a cross-platform way
317 if (model == "")
318 {
319 LLMUnitySetup.LogError("No model file provided!");
320 return null;
321 }
322 string modelPath = GetLLMManagerAssetRuntime(model);
323 if (!File.Exists(modelPath))
324 {
325 LLMUnitySetup.LogError($"File {modelPath} not found!");
326 return null;
327 }
328 string loraArgument = "";
329 foreach (string lora in lora.Trim().Split(" "))
330 {
331 if (lora == "") continue;
332 string loraPath = GetLLMManagerAssetRuntime(lora);
333 if (!File.Exists(loraPath))
334 {
335 LLMUnitySetup.LogError($"File {loraPath} not found!");
336 return null;
337 }
338 loraArgument += $" --lora \"{loraPath}\"";
339 }
340 loraManager.FromStrings(lora, loraWeights);
341
342 int numThreadsToUse = numThreads;
343 if (Application.platform == RuntimePlatform.Android && numThreads <= 0) numThreadsToUse = LLMUnitySetup.AndroidGetNumBigCores();
344
345 int slots = GetNumClients();
346 string arguments = $"-m \"{modelPath}\" -c {contextSize} -b {batchSize} --log-disable -np {slots}";
347 if (remote) arguments += $" --port {port} --host 0.0.0.0";
348 if (numThreadsToUse > 0) arguments += $" -t {numThreadsToUse}";
349 arguments += loraArgument;
350 arguments += $" -ngl {numGPULayers}";
351 if (LLMUnitySetup.FullLlamaLib && flashAttention) arguments += $" --flash-attn";
352 return arguments;
353 }

368 private void StartLLMServer(string arguments)
369 {
370 started = false;
371 failed = false;
372 bool useGPU = numGPULayers > 0;
373 LLMUnitySetup.Log($"Server command: {arguments}");
374
375 foreach (string arch in LLMLib.PossibleArchitectures(useGPU))
376 {
377 string error;
378 try
379 {
380 InitLib(arch);
381 InitService(arguments);
382 LLMUnitySetup.Log($"Using architecture: {arch}");
383 break;
384 }
385 catch (LLMException e)
386 {
387 error = e.Message;
388 Destroy();
389 }
390 catch (DestroyException)
391 {
392 break;
393 }
394 catch (Exception e)
395 {
396 error = $"{e.GetType()}: {e.Message}";
397 }
398 LLMUnitySetup.Log($"Tried architecture: {arch}, " + error);
399 }
400 if (llmlib == null)
401 {
402 LLMUnitySetup.LogError("LLM service couldn't be created");
403 failed = true;
404 return;
405 }
406 CallWithLock(StartService);
407 LLMUnitySetup.Log("LLM service created");
408 }

600 public void ApplyLoras()
601 {
602 LoraWeightRequestList loraWeightRequest = new LoraWeightRequestList();
603 loraWeightRequest.loraWeights = new List<LoraWeightRequest>();
604 float[] weights = loraManager.GetWeights();
605 for (int i = 0; i < weights.Length; i++)
606 {
607 loraWeightRequest.loraWeights.Add(new LoraWeightRequest() { id = i, scale = weights[i] });
608 }
609
610 string json = JsonUtility.ToJson(loraWeightRequest);
611 int startIndex = json.IndexOf("[");
612 int endIndex = json.LastIndexOf("]") + 1;
613 json = json.Substring(startIndex, endIndex - startIndex);
614
615 IntPtr stringWrapper = llmlib.StringWrapper_Construct();
616 llmlib.LLM_Lora_Weight(LLMObject, json, stringWrapper);
617 llmlib.StringWrapper_Delete(stringWrapper);
618 }
